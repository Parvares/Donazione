document.getElementById('donationForm').addEventListener('submit', function(event) { event.preventDefault(); const nome = document.getElementById('nome').value.trim(); const cognome = document.getElementById('cognome').value.trim(); const email = document.getElementById('email').value.trim(); const acceptCheckbox = document.getElementById('accept'); if (!nome || !cognome || !email) { alert('Per favore compila tutti i campi obbligatori.'); return; } if (!acceptCheckbox.checked) { alert('Devi accettare il disciplinare per procedere.'); return; } const formData = { nome, cognome, email, tessera: document.getElementById('tessera').value.trim(), accept: acceptCheckbox.checked, }; // Crea un foglio Excel con i dati const wb = XLSX.utils.book_new(); const ws = XLSX.utils.json_to_sheet([formData]); XLSX.utils.book_append_sheet(wb, ws, "Dati Donazione"); // Genera un file Excel e fornisci il link di download const file = XLSX.write(wb, { bookType: 'xlsx', type: 'binary' }); const blob = new Blob([s2ab(file)], { type: "application/octet-stream" }); const downloadLink = document.createElement('a'); downloadLink.href = URL.createObjectURL(blob); downloadLink.download = 'dati_donazione.xlsx'; downloadLink.click(); document.getElementById('responseMessage').classList.remove('hidden'); document.getElementById('downloadLink').classList.remove('hidden'); }); function s2ab(s) { const buf = new ArrayBuffer(s.length); const view = new Uint8Array(buf); for (let i = 0; i < s.length; i++) { view[i] = s.charCodeAt(i) & 0xFF; } return buf; }
